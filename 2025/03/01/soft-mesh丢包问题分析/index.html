soft_mesh丢包问题分析

开源鸿蒙 OpenHarmony soft_mesh Hi3863 星闪
soft_mesh
发布日期:   2025-03-01

一、前言

某位开发者遇到了这个问题长文本执行Mesh转发截断和丢失问题 · Issue #IBPR93 · Nebula-Yong/soft_mesh - Gitee.com

上文已经解决了长文本被截断的问题，篇文章将分析丢包的问题。

二、问题复现

该issue提到“很少能够转发成功，丢包现象严重”，为了方便观察，首先先修改一下广播节点的代码，如下所示，主要是添加了一个消息索引，方便观察。

void test_api_task(void)
{
    LOG("test_api_task run...\n");
    // 设置Mesh网络的SSID和密码
    if(mesh_init("FsrMesh", "12345678") != 0) {
        LOG("mesh_init failed.\n");
    } else {
        LOG("mesh_init successfully.\n");
    }
    int i = 0; // 添加的索引
    while (1)
    {
        if (mesh_network_connected() == 0) {
            LOG("Mesh network not connected.\n");
            osDelay(100);
            continue;
        }
        osDelay(100);
        char broadcast_msg[64] = {0};
        snprintf(broadcast_msg, sizeof(broadcast_msg), "0123456789|%d", i); // 加入索引传输
        if (mesh_broadcast(broadcast_msg) != 0) {
            LOG("mesh_broadcast failed.\n");
        } else {
            LOG("mesh_broadcast %d successfully.\n",i); // 打印现在是第几次广播
        }
        i++;
        char src_mac[7] = {0};
        src_mac[6] = '\0';
        char data[512] = {0};
        if(mesh_recv_data(src_mac, data) == 0) {
            LOG("Received data from client: %s, MAC: %s\n", data, src_mac);
        }else {
            LOG("no data recv.\n");
            continue;
        }
        LOG("say hi to mac: %s\n", src_mac);
        mesh_send_data(src_mac, "Hi, Node!");
    }
    
}

在四个节点的组网的情况下，的确存在很严重的丢包现象，光是节点最新的数据包的编号之间的差异就可以确定的确存在很严重的丢包现象。

把节点的数量缩小看看，先测试单纯两个节点之间的通讯，两个节点之间存在丢包现象，丢包率40%。对于这个问题，有两种可能性：

  • 本身这个WiFi通讯的能力很差，1S一个数据包就是这个丢包率
  • 自己的soft mesh代码没写好，导致丢包现象

三、丢包原因分析

我们先计算一下传10个字符要占用多少带宽。10个字符就是10Byte,网速就是10Byte/s,等于80Bps的带宽，即使加上所谓的报头也远远低于20M的理论带宽极限，所以目前的丢包问题肯定不是文本长度问题，那就先用单纯的socket通讯先进行两个节点之间的传输测试，看看同样文本下是否存在同样的丢包问题，先排除本身开发板的WiFi通讯问题。

3.1 单独Socket TCP通讯测试

int softap_sample_init(void *param)
{
    param = param;

    /* 等待wifi初始化完成 */
    while (wifi_is_wifi_inited() == 0) {

    }
    (void)memset_s(result, sizeof(result), 0, sizeof(result));
    // int retu;
    // td_u32 sta_index;
    // #define EXT_AT_MACSTR "%02x:%02x:%02x:%02x:**:**"
    // #define ext_at_mac2str(a) (a)[0], (a)[1], (a)[2], (a)[3]
    // while (1)
    // {
    //     sta_num = 6;
    //     retu = wifi_softap_get_sta_list(result, &sta_num);
    //     if (retu != ERRCODE_SUCC) {
    //         printf("get_sta_list fail");
    //     }

    //     for (sta_index = 0; sta_index < sta_num; sta_index++) {
    //         osal_printk("+SHOWSTA:"EXT_AT_MACSTR", rssi:%d, rate:%ukbps\r\n",
    //             ext_at_mac2str(result[sta_index].mac_addr), result[sta_index].rssi, result[sta_index].best_rate);
    //         osal_printk("\r\n");
    //     }
    //     osDelay(100);
    // }
    free(result);

    // tcp server代码
    #define WIFI_STA_SAMPLE_LOG "TCP_SERVER"
    #define SERVER_PORT 10086  // 服务器监听端口
    #define BACKLOG 5   // 最大连接数
    PRINT("%s::starting tcp server...\r\n", WIFI_STA_SAMPLE_LOG);
    td_s32 sfd, new_fd;
    struct sockaddr_in srv_addr, cli_addr;
    socklen_t sin_size;
    td_char recv_msg[1024];
    td_s32 ret, opt = 1;
    td_s32 tos = 128;  // 设置TOS值，与客户端匹配

    // 创建服务器socket
    sfd = socket(AF_INET, SOCK_STREAM, 0);
    if (sfd < 0) {
        uapi_at_printf("{tcp_server: socket creation failed, return is %d}\r\n", sfd);
        return EXT_ERR_FAILURE;
    }

    // 设置socket选项
    if (setsockopt(sfd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt)) < 0) {
        uapi_at_printf("{tcp_server: setsockopt SO_REUSEADDR failed}\r\n");
        closesocket(sfd);
        return EXT_ERR_FAILURE;
    }

    // 设置TOS选项
    ret = setsockopt(sfd, IPPROTO_IP, IP_TOS, &tos, sizeof(tos));
    if (ret) {
        uapi_at_printf("{tcp_server: setsockopt IP_TOS failed, return is %d}\r\n", ret);
        closesocket(sfd);
        return EXT_ERR_FAILURE;
    }

    // 配置服务器地址结构
    srv_addr.sin_family = AF_INET;
    srv_addr.sin_port = htons(SERVER_PORT);
    srv_addr.sin_addr.s_addr = INADDR_ANY;  // 监听所有IP地址
    memset(&(srv_addr.sin_zero), 0, sizeof(srv_addr.sin_zero));

    // 绑定socket
    ret = bind(sfd, (struct sockaddr *)&srv_addr, sizeof(struct sockaddr));
    if (ret < 0) {
        uapi_at_printf("{tcp_server: bind failed, return is %d}\r\n", ret);
        closesocket(sfd);
        return EXT_ERR_FAILURE;
    }

    // 监听端口
    ret = listen(sfd, BACKLOG);
    if (ret < 0) {
        uapi_at_printf("{tcp_server: listen failed, return is %d}\r\n", ret);
        closesocket(sfd);
        return EXT_ERR_FAILURE;
    }

    PRINT("%s::listening on port %d\r\n", WIFI_STA_SAMPLE_LOG, SERVER_PORT);

    // 循环等待客户端连接
    sin_size = sizeof(struct sockaddr_in);
    while (1) {
        new_fd = accept(sfd, (struct sockaddr *)&cli_addr, &sin_size);
        if (new_fd < 0) {
            uapi_at_printf("{tcp_server: accept failed, return is %d}\r\n", new_fd);
            continue;  // 出现错误时继续等待新的连接
        }

        PRINT("%s::connection received from %s\r\n", WIFI_STA_SAMPLE_LOG, inet_ntoa(cli_addr.sin_addr));

        // 接收数据
        ret = recv(new_fd, recv_msg, sizeof(recv_msg) - 1, 0);
        if (ret > 0) {
            recv_msg[ret] = '\0';  // 确保接收到的数据是字符串
            PRINT("Received message: %s\r\n", recv_msg);
        } else if (ret == 0) {
            PRINT("Client disconnected.\r\n");
        } else {
            uapi_at_printf("{tcp_server: recv failed}\r\n");
        }

        // 关闭当前连接socket
        closesocket(new_fd);
    }
    closesocket(sfd);
    PRINT("%s::server shut down.\r\n", WIFI_STA_SAMPLE_LOG);

    return 0;
}

/*****************************************************************************
  STA 匹配目标AP
*****************************************************************************/
td_s32 example_get_match_network(wifi_sta_config_stru *expected_bss)
{
    td_s32  ret;
    td_u32  num = 64; /* 64:扫描到的Wi-Fi网络数量 */
    td_char expected_ssid[] = "FsrLab_AP";
    td_char key[] = "12345678"; /* 待连接的网络接入密码 */
    td_bool find_ap = TD_FALSE;
    td_u8   bss_index;
    /* 获取扫描结果 */
    td_u32 scan_len = sizeof(wifi_scan_info_stru) * WIFI_SCAN_AP_LIMIT;
    wifi_scan_info_stru *result = osal_kmalloc(scan_len, OSAL_GFP_ATOMIC);
    if (result == TD_NULL) {
        return -1;
    }
    memset_s(result, scan_len, 0, scan_len);
    ret = wifi_sta_get_scan_info(result, &num);
    if (ret != 0) {
        osal_kfree(result);
        return -1;
    }

    /* 格式化打印扫描到的Wi-Fi网络信息，包括完整的原始信息 */
    PRINT("Scan Results: Found %d AP(s)\r\n", num);
    for (bss_index = 0; bss_index < num; bss_index++) {
        PRINT("AP %d: \r\n", bss_index + 1);
        PRINT("  SSID: %s\r\n", result[bss_index].ssid);
        PRINT("  BSSID: %02X:%02X:%02X:%02X:%02X:%02X\r\n",
              result[bss_index].bssid[0], result[bss_index].bssid[1], result[bss_index].bssid[2],
              result[bss_index].bssid[3], result[bss_index].bssid[4], result[bss_index].bssid[5]);
        PRINT("  RSSI: %d dBm\r\n", result[bss_index].rssi);
        PRINT("  Security Type: %d\r\n", result[bss_index].security_type);
        PRINT("  Band: %d\r\n", result[bss_index].band);
        PRINT("  Channel Number: %d\r\n", result[bss_index].channel_num);
    }

    /* 筛选扫描到的Wi-Fi网络，选择待连接的网络 */
    for (bss_index = 0; bss_index < num; bss_index ++) {
        if (strlen(expected_ssid) == strlen(result[bss_index].ssid)) {
            if (memcmp(expected_ssid, result[bss_index].ssid, strlen(expected_ssid)) == 0) {
                find_ap = TD_TRUE;
                break;
            }
        }
    }
    /* 未找到待连接AP,可以继续尝试扫描或者退出 */
    if (find_ap == TD_FALSE) {
        osal_kfree(result);
        return -1;
    }
    /* 找到网络后复制网络信息和接入密码 */
    if (memcpy_s(expected_bss->ssid, WIFI_MAX_SSID_LEN, expected_ssid, strlen(expected_ssid)) != 0) {
        osal_kfree(result);
        return -1;
    }
    if (memcpy_s(expected_bss->bssid, WIFI_MAC_LEN, result[bss_index].bssid, WIFI_MAC_LEN) != 0) {
        osal_kfree(result);
        return -1;
    }
    expected_bss->security_type = result[bss_index].security_type;
    if (memcpy_s(expected_bss->pre_shared_key, WIFI_MAX_SSID_LEN, key, strlen(key)) != 0) {
        osal_kfree(result);
        return -1;
    }
    expected_bss->ip_type = 1; /* 1：IP类型为动态DHCP获取 */
    osal_kfree(result);
    return 0;
}

/*****************************************************************************

td_bool example_check_connect_status(td_void)
{
    td_u8 index;
    wifi_linked_info_stru wifi_status;
    /* 获取网络连接状态，共查询5次，每次间隔500ms */
    for (index = 0; index < 5; index ++) {
        (void)osDelay(50); /* 50: 延时500ms */
        memset_s(&wifi_status, sizeof(wifi_linked_info_stru), 0, sizeof(wifi_linked_info_stru));
        if (wifi_sta_get_ap_info(&wifi_status) != 0) {
            continue;
        }
        if (wifi_status.conn_state == 1) {
            return 0; /* 连接成功退出循环 */
        }
    }
    return -1;
}

/*****************************************************************************
  STA DHCP状态查询
*****************************************************************************/
td_bool example_check_dhcp_status(struct netif *netif_p, td_u32 *wait_count)
{
    if ((ip_addr_isany(&(netif_p->ip_addr)) == 0) && (*wait_count <= WIFI_GET_IP_MAX_COUNT)) {
        /* DHCP成功 */
        PRINT("%s::STA DHCP success.\r\n", WIFI_STA_SAMPLE_LOG);
        return 0;
    }

    if (*wait_count > WIFI_GET_IP_MAX_COUNT) {
        PRINT("%s::STA DHCP timeout, try again !.\r\n", WIFI_STA_SAMPLE_LOG);
        *wait_count = 0;
        g_wifi_state = WIFI_STA_SAMPLE_INIT;
    }
    return -1;
}

            break;
        }
    }
}

    td_char ifname[WIFI_IFNAME_MAX_SIZE + 1] = "wlan0"; /* 创建的STA接口名 */
    wifi_sta_config_stru expected_bss = {0}; /* 连接请求信息 */
    struct netif *netif_p = TD_NULL;
    td_u32 wait_count = 0;

    /* 创建STA接口 */
    if (wifi_sta_enable() != 0) {
        return -1;
    }
    PRINT("%s::STA enable succ.\r\n", WIFI_STA_SAMPLE_LOG);

    do {
        // printf("g_wifi_state=%d\r\n", g_wifi_state);
        (void)osDelay(1); /* 1: 等待10ms后判断状态 */
        if (g_wifi_state == WIFI_STA_SAMPLE_INIT) {
            PRINT("%s::Scan start!\r\n", WIFI_STA_SAMPLE_LOG);
            g_wifi_state = WIFI_STA_SAMPLE_SCANING;
            /* 启动STA扫描 */
            if (wifi_sta_scan() != 0) {
                g_wifi_state = WIFI_STA_SAMPLE_INIT;
                continue;
            }
        } else if (g_wifi_state == WIFI_STA_SAMPLE_SCAN_DONE) {
            /* 获取待连接的网络 */
            if (example_get_match_network(&expected_bss) != 0) {
                PRINT("%s::Do not find AP, try again !\r\n", WIFI_STA_SAMPLE_LOG);
                g_wifi_state = WIFI_STA_SAMPLE_INIT;
                continue;
            }
            g_wifi_state = WIFI_STA_SAMPLE_FOUND_TARGET;
        } else if (g_wifi_state == WIFI_STA_SAMPLE_FOUND_TARGET) {
            PRINT("%s::Connect start.\r\n", WIFI_STA_SAMPLE_LOG);
            g_wifi_state = WIFI_STA_SAMPLE_CONNECTING;
            /* 启动连接 */
            if (wifi_sta_connect(&expected_bss) != 0) {
                g_wifi_state = WIFI_STA_SAMPLE_INIT;
                continue;
            }
        } else if (g_wifi_state == WIFI_STA_SAMPLE_CONNECT_DONE) {
            PRINT("%s::DHCP start.\r\n", WIFI_STA_SAMPLE_LOG);
            g_wifi_state = WIFI_STA_SAMPLE_GET_IP;
            netif_p = netifapi_netif_find(ifname);
            if (netif_p == TD_NULL || netifapi_dhcp_start(netif_p) != 0) {
                PRINT("%s::find netif or start DHCP fail, try again !\r\n", WIFI_STA_SAMPLE_LOG);
                g_wifi_state = WIFI_STA_SAMPLE_INIT;
                continue;
            }
        } else if (g_wifi_state == WIFI_STA_SAMPLE_GET_IP) {
            if (example_check_dhcp_status(netif_p, &wait_count) == 0) {
                break;
            }
            wait_count++;
        }
    } while (1);
    printf("%s break example_sta_function\r\n",WIFI_STA_SAMPLE_LOG);
    print_ip(netif_p);
    return 0;
}

int sta_sample_init(void *param)
{
    param = param;

    /* 注册事件回调 */
    if (wifi_register_event_cb(&wifi_event_cb) != 0) {
        PRINT("%s::wifi_event_cb register fail.\r\n", WIFI_STA_SAMPLE_LOG);
        return -1;
    }
    PRINT("%s::wifi_event_cb register succ.\r\n", WIFI_STA_SAMPLE_LOG);

    /* 等待wifi初始化完成 */
    while (wifi_is_wifi_inited() == 0) {
        (void)osDelay(10); /* 1: 等待100ms后判断状态 */
    }
    PRINT("%s::wifi init succ.\r\n", WIFI_STA_SAMPLE_LOG);

    /* 执行 WiFi 连接 */
    if (example_sta_function() != 0) {
        PRINT("%s::example_sta_function fail.\r\n", WIFI_STA_SAMPLE_LOG);
        return -1;
    }
    PRINT("%s::finish connect wifi, start tcp communication.\r\n", WIFI_STA_SAMPLE_LOG);
    int i = 0;
    while (1) {
        td_s32 ret;
        td_u32 opt = 0;
        td_s32 tos;
        td_s32 sfd = socket(AF_INET, SOCK_STREAM, 0);  // 创建socket fd
        if (sfd < 0) {
            PRINT("%s::socket creation failed, retrying...\r\n", WIFI_STA_SAMPLE_LOG);
            osDelay(500);  // 失败后等待 500ms 重新尝试
            continue;
        }

        setsockopt(sfd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
        tos = 128; /* 128:TOS设定为128，对应tid = 4，WLAN_WME_AC_VI */
        ret = setsockopt(sfd, IPPROTO_IP, IP_TOS, &tos, sizeof(tos));
        if (ret) {
            PRINT("%s::setsockopt TOPS failed, closing socket and retrying...\r\n", WIFI_STA_SAMPLE_LOG);
            closesocket(sfd);
            osDelay(500);
            continue;
        }

        struct sockaddr_in srv_addr = {0};
        const td_char *peer_ipaddr = "192.168.137.2";
        const td_u16 peer_port = 10086;
        srv_addr.sin_family = AF_INET;
        srv_addr.sin_addr.s_addr = inet_addr(peer_ipaddr);
        srv_addr.sin_port = htons(peer_port);

        ret = connect(sfd, (struct sockaddr *)&srv_addr, sizeof(srv_addr)); // 连接目标服务器
        if (ret != 0) {
            PRINT("%s::connect failed, retrying in 1s...\r\n", WIFI_STA_SAMPLE_LOG);
            closesocket(sfd);
            osDelay(1000);
            continue;
        }
        PRINT("%s::connected to server %s:%d\r\n", WIFI_STA_SAMPLE_LOG, peer_ipaddr, peer_port);

        /* 循环发送数据 */
        // TD_CONST td_char *send_msg = "0123456789|";
        char send_msg[64] = {0};
        snprintf(send_msg, sizeof(send_msg), "0123456789|%d", i);
        i++;
        td_u32 send_len = strlen(send_msg);

        ret = send(sfd, send_msg, send_len, 0);
        if (ret <= 0) {
            PRINT("%s::send failed, closing connection and retrying...\r\n", WIFI_STA_SAMPLE_LOG);
            closesocket(sfd);
            continue;;  // 退出循环，重新建立连接
        }
        PRINT("%s::sent %d bytes: %s\r\n", WIFI_STA_SAMPLE_LOG, ret, send_msg);

        /* 关闭 socket 并重新建立连接 */
        closesocket(sfd);
        osDelay(100);
    }

    return 0;
}


static void sta_sample_entry(void)
{
    osThreadAttr_t attr;
    attr.name       = "sta_sample_task";
    attr.attr_bits  = 0U;
    attr.cb_mem     = NULL;
    attr.cb_size    = 0U;
    attr.stack_mem  = NULL;
    attr.stack_size = WIFI_TASK_STACK_SIZE;
    attr.priority   = WIFI_TASK_PRIO;
    if (osThreadNew((osThreadFunc_t)sta_sample_init, NULL, &attr) == NULL) {
        PRINT("%s::Create sta_sample_task fail.\r\n", WIFI_STA_SAMPLE_LOG);
    }
    PRINT("%s::Create sta_sample_task succ.\r\n", WIFI_STA_SAMPLE_LOG);
}

/* Run the sta_sample_task. */
app_run(sta_sample_entry);

经过实际测试，单纯两个开发板之间的socket tcp通讯是不会有那么严重的丢包现象的。

3.2 分析soft_mesh部分的代码

所以基本可以确定是soft_mesh的问题，分析思路的话和之前一致，先在最底层的发送端和接收端打印一下内容，看看是哪边的问题，在广播节点加入广播序列号方便观察，代码如下。同时将application\samples\soft_mesh\hal\src\hal_wifi.c的ENABLE_LOG宏定义打开，打印底层的日志

char broadcast_msg[64] = {0};
snprintf(broadcast_msg, sizeof(broadcast_msg), "0123456789|%d", i);
if (mesh_broadcast(broadcast_msg) != 0) {
    LOG("mesh_broadcast failed.\n");
} else {
    LOG("mesh_broadcast %d successfully.\n",i);
}
i++;

很快就发现，广播函数会存在吞包的现象，也就是调用了mesh_broadcast函数但是底层并没有将这个包发出去，查看串口的发送端的日志可以看到有在创建socket套接字时失败了。同时也伴随着比较严重的延迟现象。

经过分析，其原因是MEMP_NUM_TCP_PCB和MEMP_NUM_TCP_PCB_LISTEN宏定义的值太小，导致创建新的socket的时候已经用完了资源导致无法创建新的socket。本soft_mesh项目用到的socket有用于传输心跳包的socket、接收来自其他节点的socket和发送信息的socket，理论上只需要使用3个TCP通讯，Hi3863移植的MEMP_NUM_TCP_PCB数值默认时6，似乎也远远大于理论值，但是MEMP_NUM_TCP_PCB数量包括这些，这就是为什么会导致创建socket失败的原因。

  • ESTABLISHED（已建立连接）
  • SYN-SENT（正在主动连接）
  • SYN-RECEIVED（正在被动连接）
  • CLOSE-WAIT（等待本地关闭）
  • LAST-ACK（等待最后的 ACK）
  • FIN-WAIT-1 / FIN-WAIT-2（主动关闭）
  • CLOSING（同时关闭中）

将open_source\lwip\lwip_adapter\liteos_207\src\include\lwip\lwipopts_default.h里的这两个数值调大，即可解决丢包问题。

#ifndef MEMP_NUM_TCP_PCB
#define MEMP_NUM_TCP_PCB        12
#endif

#ifndef MEMP_NUM_TCP_PCB_LISTEN
#define MEMP_NUM_TCP_PCB_LISTEN 15
#endif

3.3 延迟问题

尽管目前解决了丢包问题，但是再看串口日志的时候，发现了有比较大的延迟现象，据不精确观察，两者之间的差距是5个包的时间，也就是延迟在5秒的样子，该问题后续再分析解决，目前先观察一下多节点之间是否还会存在丢包现象。

3.4 多节点测试

接下来测试了4个节点的组网通讯，发现在现有配置下还是会存在丢包问题，于是继续加大配置如下所示，即可解决丢包问题，所以这个PCB的数量设置与预期要组网的数量相关。

#ifndef MEMP_NUM_TCP_PCB
#define MEMP_NUM_TCP_PCB        24
#endif

#ifndef MEMP_NUM_TCP_PCB_LISTEN
#define MEMP_NUM_TCP_PCB_LISTEN 30
#endif

四、总结

本文分析了Soft Mesh在多节点通信中的丢包问题，并通过实验和代码分析找到了主要的原因。

4.1 主要发现

  1. 实验复现
    • 在四个节点的组网情况下，丢包现象较为严重。
    • 在两个节点间的通讯测试中，丢包率高达40%。
    • 通过独立的Socket TCP通信测试发现，单独的Wi-Fi通讯并没有如此严重的丢包问题，说明问题可能出现在Soft Mesh的实现上。
  2. 代码分析
    • 在Soft Mesh的广播实现中，添加了广播序列号以观察数据包的发送情况。
    • 发现广播函数会出现吞包现象，即调用mesh_broadcast但底层并未真正发送数据包。
    • 通过日志分析，发现创建Socket套接字失败，导致数据无法发送。
    • 进一步分析发现，MEMP_NUM_TCP_PCB和MEMP_NUM_TCP_PCB_LISTEN宏定义的默认值过小，导致Socket资源耗尽。

4.2 解决方案

  1. 调整Socket资源配置

    • 通过增大 open_source\lwip\lwip_adapter\liteos_207\src\include\lwip\lwipopts_default.h文件下的MEMP_NUM_TCP_PCB和MEMP_NUM_TCP_PCB_LISTEN` 的值，避免Socket创建失败的问题。

    • 具体配置参考如下具体值还是需要根据实际情况进行修改

      #ifndef MEMP_NUM_TCP_PCB
#define MEMP_NUM_TCP_PCB        24
#endif

#ifndef MEMP_NUM_TCP_PCB_LISTEN
#define MEMP_NUM_TCP_PCB_LISTEN 30
#endif

    • 经过调整后，在多节点通信测试中丢包现象大幅改善。

  2. 多节点测试与验证

    • 经过调整后，在4个节点的组网情况下，Soft Mesh的丢包问题显著减少。
    • 但仍然存在一定的延迟现象，约为5秒，需要进一步优化。
    • 未来可能需要优化Mesh协议的调度机制，以减少数据传输延迟。

4.3 未来优化方向

  1. 优化延迟问题
    • 目前仍观察到约5秒的传输延迟，可能与TCP窗口大小、重传机制、队列处理等因素相关。
    • 需要进一步分析日志，优化数据发送机制。
